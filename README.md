## Install

```bash
npm install --save setaui
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'setaui'
import 'setaui/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
